# kbdiag v3 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 将 kbdiag 从诊断工具升级为 DBA 专家工具箱，新增测试框架、全局参数解析、8 项新 check、扩展 perf/locks/space，以及 sql/wait/progress/params/stat/obj/temp/watch/conf/audit/logs/remote 共 12 个新命令。

**Architecture:** 沿用 v2 模块化结构（`lib/cmd_*.sh` → `build.sh` → `dist/kbdiag`）。新增 `test/` 目录作为纯 Shell 测试框架，从本地 Mac 通过 SSH 编排两台测试 VM，VM 上零依赖。全局参数（`--top`、`--quiet`、`--format`、`--no-color`、`--timeout`）在 `lib/core.sh` 统一解析，各命令函数接收剩余参数。JSON 输出通过 `core.sh` 的 `json_*` helper 函数实现。

**Tech Stack:** Bash 4+，ksql（KingbaseES 客户端），repmgr（可选），标准 POSIX 工具（awk、df、du、column、python3）

## Global Constraints

- 必须以 `kingbase` OS 用户运行；`require_kingbase_user` 在 dispatch 前调用
- 所有 DB 查询通过 `ksql_q "<sql>"`，禁止交互会话
- KingbaseES 布尔值返回 `true`/`false`（非 `t`/`f`）：比较前必须 `tr -d '[:space:]'`
- 系统视图用 `sys_` 前缀（`sys_stat_activity`、`sys_locks` 等）；size 函数保留 `pg_` 前缀（`pg_relation_size` 等）
- WAL 目录：优先 `$KB_DATA_DIR/sys_wal`，回退 `pg_wal`
- `repmgr_available` 返回 false 时相关命令跳过并 warn，不报错
- 颜色：`[[ -t 1 ]] && NO_COLOR==""` 时启用；`NO_COLOR=1` 或非 TTY 时关闭
- `build.sh` 每个 Task 完成后必须重新运行，`dist/kbdiag` 是唯一分发产物
- 每个功能 Task 完成标准：`bash test/run_tests.sh <cmd>` 全部 PASS 才 commit
- 测试节点：node1(primary) `ssh -p 57103 kevin@127.0.0.1`；node2(standby) `ssh -p 57123 kevin@127.0.0.1`；切换用户：`sudo -i -u kingbase`
- TOP_N 默认值：10（全局），各命令可在 SQL 中用 `LIMIT $TOP_N` 应用

```bash
cat > /Users/kevin/Documents/GitHub/kbdiag/test/lib/assert.sh << 'EOF'
# Test state
TESTS_RUN=0
TESTS_PASSED=0
TESTS_FAILED=0
FAILURES=""

_pass() {
  TESTS_RUN=$(( TESTS_RUN + 1 ))
  TESTS_PASSED=$(( TESTS_PASSED + 1 ))
  echo "  [PASS] ${CURRENT_TEST:-unknown}"
}

_fail() {
  TESTS_RUN=$(( TESTS_RUN + 1 ))
  TESTS_FAILED=$(( TESTS_FAILED + 1 ))
  local msg="  [FAIL] ${CURRENT_TEST:-unknown}: $1"
  echo "$msg"
  FAILURES="${FAILURES}${msg}\n"
}

assert_contains() {
  local output="$1" pattern="$2"
  if echo "$output" | grep -qF "$pattern" 2>/dev/null; then
    _pass
  else
    _fail "expected to contain: '$pattern'\n         got: $(echo "$output" | head -5)"
  fi
}

assert_not_contains() {
  local output="$1" pattern="$2"
  if echo "$output" | grep -qF "$pattern" 2>/dev/null; then
    _fail "expected NOT to contain: '$pattern'"
  else
    _pass
  fi
}

assert_exit_code() {
  local expected="$1" actual="$2"
  if [[ "$actual" -eq "$expected" ]]; then
    _pass
  else
    _fail "expected exit code $expected, got $actual"
  fi
}

assert_json_valid() {
  local output="$1"
  if echo "$output" | python3 -m json.tool > /dev/null 2>&1; then
    _pass
  else
    _fail "invalid JSON output:\n$(echo "$output" | head -3)"
  fi
}

print_summary() {
  echo ""
  echo "Results: $TESTS_PASSED passed, $TESTS_FAILED failed (total: $TESTS_RUN)"
  if [[ $TESTS_FAILED -gt 0 ]]; then
    echo ""
    echo "Failures:"
    echo -e "$FAILURES"
    return 1
  fi
  return 0
}
EOF
```

```bash
#!/usr/bin/env bash
set -euo pipefail
SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"

source "$SCRIPT_DIR/lib/assert.sh"
source "$SCRIPT_DIR/lib/ssh_helpers.sh"

FILTER="${1:-}"

echo "==> Deploying dist/kbdiag to both nodes..."
deploy_kbdiag
echo ""

TOTAL_RUN=0; TOTAL_PASSED=0; TOTAL_FAILED=0

for case_file in "$SCRIPT_DIR/cases"/test_*.sh; do
  cmd=$(basename "$case_file" .sh | sed 's/^test_//')
  if [[ -n "$FILTER" && "$cmd" != "$FILTER" ]]; then
    continue
  fi

  echo "==> $case_file"

  # Reset counters for this file
  TESTS_RUN=0; TESTS_PASSED=0; TESTS_FAILED=0; FAILURES=""

  # Source and run all test_* functions
  source "$case_file"
  for fn in $(declare -F | awk '{print $3}' | grep '^test_'); do
    CURRENT_TEST="$fn"
    $fn
  done

  TOTAL_RUN=$(( TOTAL_RUN + TESTS_RUN ))
  TOTAL_PASSED=$(( TOTAL_PASSED + TESTS_PASSED ))
  TOTAL_FAILED=$(( TOTAL_FAILED + TESTS_FAILED ))
  echo ""
done

echo "============================================"
echo "Total: $TOTAL_PASSED passed, $TOTAL_FAILED failed / $TOTAL_RUN run"
[[ $TOTAL_FAILED -eq 0 ]] && exit 0 || exit 1
EOF
chmod +x /Users/kevin/Documents/GitHub/kbdiag/test/run_tests.sh
```

```bash
# ─── global arg parser ────────────────────────────────────────────────────────
parse_global_args() {
  ARGS=()
  while [[ $# -gt 0 ]]; do
    case "$1" in
      -v|--verbose)   VERBOSE=1 ;;
      -q|--quiet)     QUIET=1 ;;
      --no-color)     NO_COLOR=1; _init_colors ;;
      -n|--top)       TOP_N="$2"; shift ;;
      --top=*)        TOP_N="${1#--top=}" ;;
      --format)       OUTPUT_FMT="$2"; shift ;;
      --format=*)     OUTPUT_FMT="${1#--format=}" ;;
      --timeout)      KB_QUERY_TIMEOUT="$2"; shift ;;
      --timeout=*)    KB_QUERY_TIMEOUT="${1#--timeout=}" ;;
      *)              ARGS+=("$1") ;;

    shift
  done

  _JSON_ROWS=""

json_item() {
  # json_item <name> <status> <value> [detail]
  local name="$1" status="$2" value="$3" detail="${4:-}"
  local entry="{\"name\":\"$name\",\"status\":\"$status\",\"value\":\"$(echo "$value" | sed 's/"/\\"/g')\",\"detail\":\"$(echo "$detail" | sed 's/"/\\"/g')\"}"
  _JSON_ITEMS="${_JSON_ITEMS:+${_JSON_ITEMS},}${entry}"
}

json_row() {
  # json_row key1=val1 key2=val2 ...
  local obj="{" first=1
  for pair in "$@"; do
    local k="${pair%%=*}" v="${pair#*=}"
    [[ $first -eq 0 ]] && obj="${obj},"
    obj="${obj}\"$k\":\"$(echo "$v" | sed 's/"/\\"/g')\""
    first=0
  done
  obj="${obj}}"
  _JSON_ROWS="${_JSON_ROWS:+${_JSON_ROWS},}${obj}"
}

json_end() {
  local ts; ts=$(date -u +%Y-%m-%dT%H:%M:%SZ)
  echo "{\"command\":\"${_JSON_CMD}\",\"timestamp\":\"${ts}\",\"items\":[${_JSON_ITEMS}],\"rows\":[${_JSON_ROWS}]}"
  _JSON_ITEMS=""; _JSON_ROWS=""
}
```

```sql
SELECT wait_event_type, wait_event, count(*) AS cnt,
       round(avg(EXTRACT(EPOCH FROM (now()-query_start))),1) AS avg_sec
FROM sys_stat_activity

-- perf io：表 I/O 热点

       heap_blks_read, heap_blks_hit,
       CASE WHEN (heap_blks_read+heap_blks_hit)=0 THEN 100
            ELSE round(100.0*heap_blks_hit/(heap_blks_read+heap_blks_hit),1) END AS hit_pct
FROM sys_statio_user_tables
ORDER BY heap_blks_read DESC LIMIT $TOP_N;

-- perf wal：WAL 和 checkpoint 统计
SELECT checkpoints_timed, checkpoints_req,
       pg_size_pretty(buffers_checkpoint*8192::bigint) AS ckpt_bytes,
       pg_size_pretty(buffers_clean*8192::bigint) AS bgwriter_bytes,
       pg_size_pretty(buffers_backend*8192::bigint) AS backend_bytes
FROM sys_stat_bgwriter;

-- perf top buffers
SELECT pid, usename, left(query,60) AS query,
       shared_blks_hit + shared_blks_read AS total_buffers
FROM sys_stat_activity
WHERE state='active' AND pid <> sys_backend_pid()
ORDER BY total_buffers DESC NULLS LAST LIMIT $TOP_N;

-- perf vacuum 阈值（增强）
SELECT relname, n_dead_tup,
       (SELECT setting::int FROM sys_settings WHERE name='autovacuum_vacuum_threshold') +
       (SELECT setting::float FROM sys_settings WHERE name='autovacuum_vacuum_scale_factor') * reltuples AS threshold,
       round(100.0 * n_dead_tup /
         NULLIF((SELECT setting::int FROM sys_settings WHERE name='autovacuum_vacuum_threshold') +
         (SELECT setting::float FROM sys_settings WHERE name='autovacuum_vacuum_scale_factor') * reltuples, 0), 1) AS pct
FROM sys_stat_user_tables WHERE n_dead_tup > 100 ORDER BY pct DESC NULLS LAST LIMIT $TOP_N;
```

```bash
cmd_stat() {
  local interval=5 count=1
  local arg
  for arg in "$@"; do
    case "$arg" in
      --interval=*) interval="${arg#--interval=}" ;;
      --interval) ;;  # next arg handled below
      --count=*)  count="${arg#--count=}" ;;
    esac
  done

  hdr "Instance throughput (${interval}s sample)"

  local i=0
  while [[ $i -lt $count ]]; do
    local snap1 snap2
    snap1=$(ksql_q "SELECT xact_commit, xact_rollback, blks_read, blks_hit, temp_bytes, deadlocks FROM sys_stat_database WHERE datname=current_database();")
    sleep "$interval"
    snap2=$(ksql_q "SELECT xact_commit, xact_rollback, blks_read, blks_hit, temp_bytes, deadlocks FROM sys_stat_database WHERE datname=current_database();")

    # Parse and compute deltas (awk)
    echo "$snap1\n$snap2" | awk -F'|' -v interval="$interval" '
      NR==1 { c1=$1; r1=$2; br1=$3; bh1=$4; t1=$5; d1=$6 }
      NR==2 { c2=$1; r2=$2; br2=$3; bh2=$4; t2=$5; d2=$6
        printf "TPS commits/s:   %.1f\n", (c2-c1)/interval
        printf "TPS rollbacks/s: %.1f\n", (r2-r1)/interval
        total_blk = (bh2-bh1) + (br2-br1)
        hit_pct = (total_blk > 0) ? 100.0*(bh2-bh1)/total_blk : 100
        printf "Buffer hit rate: %.1f%%\n", hit_pct
        printf "Temp bytes/s:    %.0f\n", (t2-t1)/interval
        printf "Deadlocks/s:     %.2f\n", (d2-d1)/interval
      }'

    i=$(( i + 1 ))
    [[ $i -lt $count ]] && sleep 0
  done
}
```

```bash
cmd_obj() {
  local target="${1:-}"
  if [[ -z "$target" ]]; then
    echo "Usage: kbdiag obj <schema.table>" >&2; return 1
  fi

  local schema table
  if echo "$target" | grep -q '\.'; then
    schema="${target%%.*}"; table="${target#*.}"
  else
    schema="public"; table="$target"
  fi

  hdr "Object analysis: $schema.$table"

  # 1. 基本信息
  info "Size & tuples:"
  ksql_q "
    SELECT pg_size_pretty(pg_total_relation_size('$schema.$table')) AS total_size,
           pg_size_pretty(pg_relation_size('$schema.$table')) AS table_size,
           pg_size_pretty(pg_indexes_size('$schema.$table')) AS index_size,
           n_live_tup, n_dead_tup,
           round(100.0*n_dead_tup/NULLIF(n_live_tup+n_dead_tup,0),1) AS dead_pct
    FROM sys_stat_user_tables WHERE schemaname='$schema' AND relname='$table';" | column -t -s '|' || true

  # 2. vacuum 状态
  info "Vacuum/Analyze status:"
  ksql_q "
    SELECT last_vacuum::date, last_autovacuum::date, last_analyze::date, last_autoanalyze::date,
           n_mod_since_analyze AS mods_since_analyze
    FROM sys_stat_user_tables WHERE schemaname='$schema' AND relname='$table';" | column -t -s '|' || true

  # 3. 索引
  info "Indexes (usage):"
  ksql_q "
    SELECT indexrelname, pg_size_pretty(pg_relation_size(indexrelid)) AS size,
           idx_scan, idx_tup_read, idx_tup_fetch,
           CASE WHEN indisunique THEN 'UNIQUE' ELSE '' END AS flags
    FROM sys_stat_user_indexes i
    JOIN sys_index si ON si.indexrelid = i.indexrelid
    WHERE schemaname='$schema' AND relname='$table'
    ORDER BY idx_scan DESC;" | column -t -s '|' || true

  # 4. 约束
  info "Constraints:"
  ksql_q "
    SELECT conname, contype,
           CASE contype WHEN 'p' THEN 'PRIMARY KEY' WHEN 'f' THEN 'FOREIGN KEY'
                        WHEN 'u' THEN 'UNIQUE' WHEN 'c' THEN 'CHECK' END AS type
    FROM sys_constraint WHERE conrelid = '$schema.$table'::regclass;" | column -t -s '|' || true

  # 5. verbose: 列统计
  if [[ -n "$VERBOSE" ]]; then
    info "Column statistics:"
    ksql_q "
      SELECT attname, null_frac, n_distinct, most_common_vals
      FROM sys_stats WHERE schemaname='$schema' AND tablename='$table'
      ORDER BY attname LIMIT $TOP_N;" | column -t -s '|' || true
  fi
```

```bash
cmd_logs() {
  local log_file="" since="" level="all"
  while [[ $# -gt 0 ]]; do
    case "$1" in
      --file)    log_file="$2"; shift ;;
      --file=*)  log_file="${1#--file=}" ;;
      --since)   since="$2"; shift ;;
      --since=*) since="${1#--since=}" ;;
      --level)   level="$2"; shift ;;
      --level=*) level="${1#--level=}" ;;
    esac
    shift
  done

  # 自动探测日志路径
  if [[ -z "$log_file" ]]; then
    local log_dir log_fn
    log_dir=$(ksql_q "SELECT setting FROM sys_settings WHERE name='log_directory';" | tr -d '[:space:]')
    log_fn=$(ksql_q "SELECT setting FROM sys_settings WHERE name='log_filename';" | tr -d '[:space:]')
    [[ "${log_dir}" != /* ]] && log_dir="$KB_DATA_DIR/$log_dir"
    log_file=$(ls -t "${log_dir}/"*.log 2>/dev/null | head -1 || true)
  fi

  if [[ -z "$log_file" || ! -r "$log_file" ]]; then
    warn "Log file not found or not readable. Set KB_LOG_FILE or use --file."
    return 0
  fi

  hdr "Log analysis: $log_file"

  local content="$log_file"
  # --since 过滤：用 awk 按时间戳过滤（格式 YYYY-MM-DD HH:MM:SS）
  if [[ -n "$since" ]]; then
    local mins=0
    [[ "$since" =~ ([0-9]+)h ]] && mins=$(( mins + ${BASH_REMATCH[1]} * 60 ))
    [[ "$since" =~ ([0-9]+)m ]] && mins=$(( mins + ${BASH_REMATCH[1]} ))
    local cutoff; cutoff=$(date -d "-${mins} minutes" '+%Y-%m-%d %H:%M:%S' 2>/dev/null || \
                           date -v "-${mins}M" '+%Y-%m-%d %H:%M:%S' 2>/dev/null)
    content=<(awk -v cut="$cutoff" '$0 >= cut' "$log_file")
  fi

  # 慢查询 Top N
  info "Slow queries (> ${KB_SLOW_THRESHOLD}s) Top $TOP_N:"
  grep -E "duration: [0-9]+\.[0-9]+ ms" "$log_file" 2>/dev/null | \
    awk -F'duration: ' '{split($2,a," "); if(a[1]+0 > '"$KB_SLOW_THRESHOLD"'*1000) print a[1]" ms: "$0}' | \
    sort -rn | head -"$TOP_N" || info "No slow queries found"

  # ERROR/FATAL 统计
  info "ERROR/FATAL by hour:"
  grep -E " ERROR:| FATAL:" "$log_file" 2>/dev/null | \
    awk '{match($0,/[0-9]{4}-[0-9]{2}-[0-9]{2} [0-9]{2}/,a); h[a[0]]++} END {for(k in h) print k,h[k]}' | \
    sort | tail -24 || true

  # 最频繁错误类型
  info "Top $TOP_N error messages:"
  grep -E " ERROR:| FATAL:" "$log_file" 2>/dev/null | \
    sed 's/.*ERROR: //' | sort | uniq -c | sort -rn | head -"$TOP_N" || true
}
```

```bash
cmd_remote() {
  local nodes_arg="${1:-}"; shift
  local rcmd="${1:-}"; shift
  local rargs=("$@")

  if [[ -z "$nodes_arg" || -z "$rcmd" ]]; then
    echo "Usage: kbdiag remote <node1,node2|--all> <command> [args]" >&2; return 1
  fi

  # 解析节点列表
  local nodes=()
  if [[ "$nodes_arg" == "--all" ]]; then
    # 从 repmgr 获取
    if repmgr_available; then
      while IFS= read -r line; do
        local host; host=$(echo "$line" | grep -oP 'host=\K[^ ]+' || true)
        [[ -n "$host" ]] && nodes+=("$host")
      done < <("$REPMGR" -f "$KB_REPMGR_CONF" cluster show 2>/dev/null | grep -E 'primary|standby')
    fi
    # 兜底：读 ~/.kbdiag/nodes.conf
    local conf="$HOME/.kbdiag/nodes.conf"
    if [[ ${#nodes[@]} -eq 0 && -f "$conf" ]]; then
      while IFS= read -r line; do
        [[ "$line" =~ ^# ]] && continue
        nodes+=("$line")
      done < "$conf"
    fi
  else
    IFS=',' read -ra nodes <<< "$nodes_arg"
  fi

  if [[ ${#nodes[@]} -eq 0 ]]; then
    warn "No nodes found. Use repmgr or create ~/.kbdiag/nodes.conf"
    return 0
  fi

  # 串行 SSH 执行
  for node in "${nodes[@]}"; do
    echo ""
    echo "=== [$node] ==="
    ssh -o StrictHostKeyChecking=no -o ConnectTimeout=5 "$node" \
      "sudo -i -u kingbase /tmp/kbdiag $rcmd ${rargs[*]+"${rargs[*]}"}" 2>/dev/null || \
      warn "[$node] connection failed or kbdiag not deployed"
  done
}
```
